1.
# Вывести список всех файлов в папке html
cd /var/www/html # переходим в папку html
ls -a # выводим список всех файлов в папке html с флагом -a для вывода скрытых файлов

2.
# Находим в файле access.log все упоминания строки "404"
cd ../logs/access.log # переходим в папку logs
grep 404 access.log # Находим в файле access.log все упоминания строки "404"

3.
# На основании файла 404.html создаем файл 403.html заменив в его содержимом все упоминания 404 на 403
cd ../html # переходим в папку html
cp 404.html 403.html # создаем файл 403.html на основании 404.html
sed -i "s/404/403/g" 403.html # заменяем все упоминания 404 на 403 в файле 403.html с флагом -i для создания резеврной копии файла перед редактированием

4.
# Модифицировать права доступа и владельца файла index.html следующим образом: 
Текущий пользователь должен иметь права на чтение/запись Группа www-data на чтение Все остальные не имеют доступа к файлу
chmod u+rw index.html # даем права на чтение и запись пользователю
chmod www-data+r index.html # даем право на чтение группе
chmod o-rwx index.html # убираем все права у остальных

5.
# Сделайте символическую ссылку на файл access.log в папке html
ln -s ../logs/acces.log symblink # использую флаг -s для создания символической ссылки в папке html

6.
# Выведите на экран все файлы в папке var
cd ../ # переходим в папку var
find ./var # выводим на экран все файлы в папке var

    </head>
